1. Query all translations of a namespace

Map:

function (doc) {
    var i, combinedNs;
    if (doc.namespace) {
        combinedNs = '';
        for (i = 0; i < doc.namespace.length; i++) {
            combinedNs = combinedNs + doc.namespace[i];
            emit(combinedNs, doc);
            combinedNs = combinedNs + '/'
        }
    }
}
